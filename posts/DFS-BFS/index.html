    '2': {'3', '7', '8'},

    '4': {'5', '7', '8'},
    '5': {'4', '8'},

    '7': {'1', '2', '3', '4', '6', '8'},
    '8': {'2', '4', '5', '6', '7'}
}

깊이우선탐색(DFS)

1
2
3
4
5
6
7
8

target = ['1']
visit = ['1']
while target:
    x = target.pop()
    for i in graph[x]:
        if not i in visit:
            visit.append(i)
            target.append(i)

너비우선탐색(BFS)

target = ['1']
visit = ['1']
while target:
    x = target.pop()
    for i in graph[x]:
        if not i in visit:
            visit.append(i)
            target.insert(0, i)
